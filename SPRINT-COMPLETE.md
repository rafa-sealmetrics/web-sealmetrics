## Post-deploy verification commands

```sh
# 1. HTTPS enforced
curl -sI http://sealmetrics.com/ | head -3
# Expected: HTTP/1.1 301 Moved Permanently

# 2. Security headers present (6 expected)
curl -sI https://sealmetrics.com/ | grep -iE "strict-transport|x-content-type|referrer-policy|permissions-policy|x-frame|content-security"

# 3. Pixel is production
curl -s https://sealmetrics.com/ | grep -oE "pixel(-pre)?\.sealmetrics\.com" | sort -u
# Expected: pixel.sealmetrics.com

# 4. New routes live
for u in /audit/ /case-studies/dreamplace-hotels/ /case-studies/palladium-hotel-group/ /es/glossary/cookieless-analytics/ /es/case-studies/palladium-hotel-group/; do
  echo "$u → $(curl -s -o /dev/null -w "%{http_code}" "https://sealmetrics.com$u")"
done
# Expected: 200 200 200 200 200

# 5. Sitemap trailing-slash compliance
curl -s https://sealmetrics.com/sitemap.xml | grep -c "/</loc>"
# Expected: 128

# 6. Static asset cache
CHUNK=$(curl -s https://sealmetrics.com/ | grep -oE "/_next/static/chunks/[a-f0-9]+\.js" | head -1)
curl -sI "https://sealmetrics.com$CHUNK" | grep -i cache-control
# Expected: cache-control: public, max-age=31536000, immutable

# 7. Per-post OG image
curl -sI https://sealmetrics.com/og/blog/cookieless-analytics-explained.png
# Expected: HTTP/2 200, content-type: image/png

# 8. AVIF served
curl -sI https://sealmetrics.com/logos/logo-sealmetrics-negro.avif
# Expected: HTTP/2 200, content-type: image/avif
```
